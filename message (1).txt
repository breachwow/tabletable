        function addConfig() {
            const input = document.getElementById('config-name-input');
            const name = input.value.trim();
            if (!name) return;

            const newConfig = document.createElement('div');
            newConfig.className = "bg-white/5 p-4 rounded-lg border border-white/5 hover:border-white/10 transition cursor-pointer";
            newConfig.innerHTML = `
                <h4 class="font-bold text-sm text-white">${name}</h4>

            `;

            library.prepend(newConfig);
            input.value = "";
            countEl.innerText = `${currentCount + 1}/10`;
        }

            // Start Timer
            countdownValue = 15;
            const bar = document.getElementById('timer-bar');
            const text = document.getElementById('countdown-text');
            const total = 377; // stroke-dasharray

            timerInterval = setInterval(() => {
                countdownValue--;
                text.innerText = countdownValue;
                
                const offset = total - (total * (15 - countdownValue) / 15);
                bar.style.strokeDashoffset = total - offset;

                if(countdownValue <= 0) {
                    clearInterval(timerInterval);
                    finalizePurchase(note);
                }
            }, 1000);
        }

        async function sendWebhook(title, description) {
            if(!DISCORD_WEBHOOK_URL || DISCORD_WEBHOOK_URL === "") return;

            const payload = {
                embeds: [{
                    title: title,
                    description: description,
                    color: 14833741, // #e2584d in decimal
                    timestamp: new Date().toISOString(),
                    footer: { text: "Ledger Dashboard Logs" }
                }]
            };

            try {
                await fetch(DISCORD_WEBHOOK_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
            } catch (e) {
                console.error("Webhook failed:", e);
            }
        }

        function copyToClipboard(text) {
            const el = document.createElement('textarea');
            el.value = text;
            document.body.appendChild(el);
            el.select();
            document.execCommand('copy');
            document.body.removeChild(el);
        }
